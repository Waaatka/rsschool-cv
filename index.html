function indexEqualsValue(a) {
  let start = 0;
  let end = a.length - 1;
  while (start <= end) {
      const k = (start + end) >> 1;
      if (a[k] < k) {
          start = k + 1;
      } else {
          end = k - 1;
      }
  }
  return start === a[start] ? start : -1;
}
            
          
            
const rotateMatrix = (arr) => {
  const result = [];
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr[i].length; j++) {
      if (!result[j]) {
        result[j] = [];
      }
      result[j][i] = arr[i][arr[i].length - j - 1];
    }
  }
  return result;
};
